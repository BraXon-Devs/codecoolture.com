# React: Slots

## Problem

```tsx
type NavProps = {
  cta: { text: string; href: string };
  menuItems: { text: string; href: string }[];
  title: string;
};

export const Nav = ({ cta, menuItems, title }: NavProps) => {
  return (
    <nav>
      <h1>{title}</h1>

      <ul>
        {menuItems.map((item) => (
          <li>
            <a href={item.href}>{item.text}</a>
          </li>
        ))}
      </ul>

      <a href={cta.href}>{cta.text}</a>
    </nav>
  );
};

// Usage
<Nav
  cta={{ text: "Buy the book!", href: "https://url.to/buy-my-book" }}
  menuItems={[
    { text: "Home", href: "/home" },
    { text: "About", href: "/about" },
    { text: "Contact", href: "/contact" },
  ]}
  title="My Site"
/>;
```

### v2

```tsx
type NavProps = {
  cta: React.ReactNode;
  menuItems: React.ReactNode;
  title: React.ReactNode;
};

export const Nav = ({ cta, menuItems, title }: NavProps) => {
  return (
    <nav>
      {title}

      <ul>{menuItems}</ul>

      {cta}
    </nav>
  );
};

// Usage
<Nav
  cta={<a href="https://url.to/buy-my-book">Buy the book!</a>}
  menuItems={
    <>
      <li>
        <a href="/home">Home</a>
      </li>

      <li>
        <a href="/about">About</a>
      </li>

      <li>
        <a href="/contact">Contact</a>
      </li>
    </>
  }
  title={<h1>My Site</h1>}
/>;
```

## Possible solution

```tsx
type NavProps = { children: React.ReactNode };

const Nav = ({ children }: NavProps) => {
  return <nav>{children}</nav>;
};

type TitleProps = { children: React.ReactNode };

const Title = ({ children }: TitleProps) => {
  return <h1>{children}</h1>;
};

type MenuProps = { children: React.ReactNode };

const Menu = ({ children }: MenuProps) => {
  return <ul>{children}</ul>;
};

type MenuItemProps = { children: React.ReactNode; href: string };

Menu.Item = ({ children, href }: MenuItemProps) => {
  return (
    <li>
      <a href={href}>{children}</a>
    </li>
  );
};

type CallToActionProps = { children: React.ReactNode; href: string };

const CallToAction = ({ children, href }: CallToActionProps) => {
  return <a href={href}>{children}</a>;
};

Nav.Title = Title;
Nav.Menu = Menu;
Nav.CallToAction = CallToAction;

export { Nav };
```

That will be used as the following:

```tsx
<Nav>
  <Nav.Title>My Site</Nav.Title>

  <Nav.Menu>
    <Nav.Menu.Item href="/">Home</Nav.Menu.Item>
    <Nav.Menu.Item href="/about">About</Nav.Menu.Item>
    <Nav.Menu.Item href="/contact">Contact</Nav.Menu.Item>
  </Nav.Menu>

  <Nav.CallToAction href="https://url.to/buy-my-book">
    Buy the book!
  </Nav.CallToAction>
</Nav>
```
